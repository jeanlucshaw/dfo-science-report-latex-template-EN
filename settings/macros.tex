\newcommand{\tablenotec}[2]{\textcolor{white}{$^#2$}#1$^#2$}


% Insert unnumbered section and content if parameter 2 supplied.
%
% usage: \dfostarsection{Title}{file path and name (.tex)}
%
\makeatletter
\newcommand{\dfostarsection}[2]{%

  \refstepcounter{section}%
  \section*{\MakeUppercase{#1}}%
  \addtocounter{section}{-1}%
  \def\@currentlabelname{#1}%
  \def\@currentlabel{\lowercase{\thesection}}%
  \label{sec:#1}%
  \addcontentsline{toc}{section}{\texorpdfstring{\uppercase{#1}}{#1}}

  % Insert contents of #2.tex after subsection heading
  \ifx\empty#2\empty
  \else
  \input{#2}
  \fi
}
\makeatother

% Insert bibliography heading as document section
%
% usage: \dfobibliographysection{References page title}
%
\newcommand{\dfobibliographysection}[1]{

    \ifx\empty#1\empty
        \dfostarsection{\refname}{}
        \renewcommand{\refname}{}
    \else
        \dfostarsection{#1}{}
        \renewcommand{\refname}{}
    \fi
    \vspace{-0.75cm}
}
